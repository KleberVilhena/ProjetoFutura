Exemplo de Pseudocódigo
ALGORITMO "VerificarAprovacao"

INÍCIO
    // Declaração da variável
    DECLARE nota: REAL

    // Entrada de dados
    ESCREVA "Digite a nota do aluno: "
    LEIA nota

    // Decisão (Lógica principal)
    SE nota >= 7 ENTÃO
        ESCREVA "Aluno APROVADO!"
    SENÃO
        ESCREVA "Aluno REPROVADO."
    FIM_SE

FIM
